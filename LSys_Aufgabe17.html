  1,2 + 2,4y = x   (1)

x = 1,2 + 2,4y in (2) eingesetzt 
 
3y + 2(1,2 + 2,4y) + 21 = 0 
 
   3y + 2,4 + 4,8y + 21 = 0 
 
           7,8 y + 23,4 = 0 |- 23,4 
 
                   7,8y = - 23,4 |:7,8 
 
                      y = -3
